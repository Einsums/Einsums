if(EINSUMS_WITH_EXAMPLES)
  einsums_add_pseudo_target(Examples.Modules.Einsums.Version)
  einsums_add_pseudo_dependencies(Examples.Modules Examples.Modules.Einsums.Version)
  if(EINSUMS_WITH_TESTS AND EINSUMS_WITH_TESTS_EXAMPLES)
    einsums_add_pseudo_target(Tests.Examples.Modules.Einsums.Version)
    einsums_add_pseudo_dependencies(Tests.Examples.Modules Tests.Examples.Modules.Einsums.Version)
  endif()
else()
  return()
endif()

set(example_programs EinsumsInfo)

foreach(example_program ${example_programs})
  set(sources ${example_program}.cpp)

  source_group("Source Files" FILES ${sources})

  # add example executable
  einsums_add_executable(
    ${example_program} INTERNAL_FLAGS
    SOURCES ${sources} ${${example_program}_FLAGS}
    FOLDER "Examples/Modules/Einsums/Version"
    NOINSTALL
  )

  einsums_add_example_target_dependencies("Modules.Einsums.Version" ${example_program})

  if(EINSUMS_WITH_TESTS AND EINSUMS_WITH_TESTS_EXAMPLES)
    einsums_add_example_test(
      "Modules.Einsums.Version" ${example_program} ${${example_program}_PARAMETERS}
    )
  endif()
endforeach()
